立即执行函数（IIFE）是指在 函数声明后立即执行的函数表达式。它会形成一个独立的作用域，外部无法访问内部私有属性。

语法为使用括号将函数包裹并跟一个小括号，小括号在括号内外都可。

<JSRunner :code='code1'/>

<script setup>
const code1 = `// 命名函数表达式 + 立即执行
(function foo() {
  console.log('立即执行');
})();

// 或者这样
(function bar() {
  console.log('立即执行');
}());

// 传递参数
(function foo(a) {
  console.log(a); // 1
})(1)

// 错误，foo() undefined
// foo() 

const request = ( function () {
  const private = 'private';
  const public = 'public';
  return { public }
})()
console.log(request.public) // "public"
console.log(request.private) // undefined
`
</script>
